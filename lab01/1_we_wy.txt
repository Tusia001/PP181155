#include <stdio.h>

int main() {
    double a = 1.0e16;
    double b = -1.0e16;
    double c = 1.0;

    double wynik1 = (a + b) + c;
    double wynik2 = a + (b + c);


    printf("Demonstracja niełączności dodawania liczb zmiennoprzecinkowych:\n");
    printf("(a + b) + c = (%.1e + %.1e) + %.1f = %.1f\n", a, b, c, wynik1);
    printf("a + (b + c) = %.1e + (%.1e + %.1f) = %.1f\n", a, b, c, wynik2);
    return 0;
}
